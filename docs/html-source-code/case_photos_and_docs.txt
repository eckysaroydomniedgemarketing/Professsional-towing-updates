  function uploadDocument(){
    const MAX_FILE_SIZE = 7 * 1024 * 1024; // 7MB - Number needs to be in bytes, hence the multiplication

    documentfieldsToUpload = new Array();
    invalid_ext = false;
    size_too_big = false;

    $('#module_25 input[type=file], #module_25 select').each(
      function(){
         if( $(this).is("input[type=file]") ){
          var extension = $(this).val().split('.').pop().toLowerCase();

          if (!(extension in {jpg:1, jpeg:1, png:1, gif:1, pdf:1, doc:1, docx:1, xls:1, xlsx:1, csv:1, msg:1, txt:1, tif:1, rtf:1})) {
            invalid_ext = true;
            $(this).addClass('is-invalid');
          }

          if (this.files[0].size > MAX_FILE_SIZE)
          {
            size_too_big = true;
            $(this).addClass('is-invalid');
          }
        }

        if($(this).val()){
          documentfieldsToUpload[documentfieldsToUpload.length] = $(this);
        }
      }
    );

    if (invalid_ext) {
      rdn_alert('documents_alerts', 'Files must be rtf, pdf, doc, docx, xls, csv, msg, txt, jpg, jpeg, png, gif or tif', null, null, null, true);
    } else if (size_too_big) {
      rdn_alert('documents_alerts', 'File upload failed. The file you are attempting to upload exceeds the size limit (7.2MB)', null, null, null, true);
    } else {
      continueUploadDoc();
    }
  }
