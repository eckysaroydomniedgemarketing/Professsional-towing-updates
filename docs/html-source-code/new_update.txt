        "use strict";

        function toggleSelectAll(el, type) {
            var text = $(el).html().trim();
            if (text === "Select All") {
                $('input.js-' + type + '-update-check').prop('checked', true);
                $(el).html('Unselect All');
            } else {
                $('input.js-' + type + '-update-check').prop('checked', false);
                $(el).html('Select All');
            }
        }

        function removeUpdates(type, updates) {
            var form = $('#js-remove-update-form');
            var actionUrl = form.attr('action');
            var formData = form.serializeArray();
            for (var i = 0; i < updates.length; i++) {
                formData.push({ name: "update_id[]", value: updates[i] });
            }
            $.post(actionUrl, formData, function() {
                var numberOfUserOwnedUpdates = 0;
                for (var i = 0; i < updates.length; i++) {
                    var updateElement = $('.js-' + type + '-update-' + updates[i]);
                    if (updateElement.hasClass('js-user-owned')) numberOfUserOwnedUpdates++;
                    updateElement.remove();
                }
                if (window !== window.parent) parent.decrementNewUpdateCounterBadge(numberOfUserOwnedUpdates);
                RDN.toaster.toast('Updates were removed.');
            }).fail(function() {
                RDN.toaster.toast('Unable to remove updates.', true);
            });
        }

        function removeSelected(type) {
            var updates = $('input.js-' + type + '-update-check:checked').map(function() {
                return $(this).val();
            }).get();
            if (!updates.length) {
                RDN.toaster.toast('No updates are selected.', true);
                return;
            }
            if(confirm('Are you sure you wish to delete these ' + updates.length + ' updates? This cannot be undone.')) {
                removeUpdates(type, updates);
            }
        }

            });

            $(".js-remove-selected-new").click(function () {
                removeSelected("new");
            });
            $(".js-remove-selected-inv").click(function () {
                removeSelected("inv");
            });

            $(".js-remove-new").click(function() {
                var updateId = $(this).data("id");

            });
            $(".js-remove-inv").click(function() {
                var updateId = $(this).data("id");
