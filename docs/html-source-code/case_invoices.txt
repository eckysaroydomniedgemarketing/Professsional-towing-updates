function handleInternalStatusChange(selectElement, internal_default_val, feeTypeLabel) {
    var originalValue = selectElement.getAttribute('data-original-value');
    
    // if nonexistent/empty, means its technically a new request, so set to 1
    if (!originalValue || originalValue === '') {
        originalValue = '1';
    }
    
    selectElement.setAttribute('data-changed-original', originalValue);    
    var feeId = selectElement.getAttribute('data-fee-id');
    
    var modalCloseHandler = function() {
        var storedOriginal = selectElement.getAttribute('data-changed-original');
        if (storedOriginal) {
            selectElement.value = storedOriginal;
            selectElement.removeAttribute('data-changed-original');
        }
        $(document).off('hidden.bs.modal.internalStatusRestore');
    };

    $(document).on('hidden.bs.modal.internalStatusRestore', '.modal', modalCloseHandler);    
    
    var existingComment = selectElement.getAttribute('data-existing-comment') || '';
    openInternalModal(feeId, selectElement.value, null, null, existingComment, internal_default_val, feeTypeLabel);
}
